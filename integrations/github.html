display all open issues

commands.open_issues = function() { github.issues.repoIssues({ "user": user, "repo": repo, "state": "open" }, function(err, issues) { if (err) { console.dir(err) }; print_issues(issues, "No open issues"); }); };

constructor

The constructor for this module takes a hash of settings values, and prepares the instantiated module for communicating with github. The token can be generated and stored by the setup method.

  • settings: Object
    • token: String a Github oauth authentication token
    • user: String the Github user which holds the repository for this project
    • repo: String the name of the repository for this project
module.exports = function(settings) { if (settings) { if (settings.token) { github.authenticate({ "type": "oauth", "token": settings.token }); authenticated = true; } user = settings.user; repo = settings.repo; } return { "commands": commands, "opts": {} }; };
